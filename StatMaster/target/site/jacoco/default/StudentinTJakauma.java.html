    public String yksiSuuntainen(int df, double t) {
        if (t < jakaumataulu[df][0]) {
            return "ns";
        } else if (t >= jakaumataulu[df][0] && t < jakaumataulu[df][2]) {
            return "p <0.05";
        } else if (t >= jakaumataulu[df][2] && t < jakaumataulu[df][5]) {
            return "p <0.01";
        }
        return "p <0.001";
    }
    
    public String kaksiSuuntainen(int df, double t) {
        if (t < jakaumataulu[df][1]) {
            return "ns";
        } else if (t >= jakaumataulu[df][1] && t < jakaumataulu[df][3]) {
            return "p <0.05";
        } else if (t >= jakaumataulu[df][3] && t < jakaumataulu[df][6]) {
            return "p <0.01";
        }
        return "p <0.001";
    }
